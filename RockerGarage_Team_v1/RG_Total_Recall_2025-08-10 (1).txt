        const a = document.getElementById('rg-intro'); let started=false;
        function go(){ if(started) return; started=true; a.volume=0.6; a.play().catch(()=>{}); }
        ['pointerdown','keydown','touchstart'].forEach(e=>window.addEventListener(e,go,{once:true}));
        document.addEventListener('visibilitychange',()=>{ if(document.hidden) a.pause(); else if(started) a.play().catch(()=>{}); });
      })();
    </script>
